def draw_neural_net(ax, left, right, bottom, top, layer_sizes):
    n_layers = len(layer_sizes)
    v_spacing = (top - bottom)/float(max(layer_sizes))
    h_spacing = (right - left)/float(len(layer_sizes) - 1)
    # Nodes
    for n, layer_size in enumerate(layer_sizes):
        layer_top = v_spacing*(layer_size - 1)/2. + (top + bottom)/2.
        for m in range(layer_size):
            circle = plt.Circle((n*h_spacing + left, layer_top - m*v_spacing), v_spacing/4.,
                                color='w', ec='k', zorder=4)
            ax.add_artist(circle)
    # Edges
    for n, (layer_size_a, layer_size_b) in enumerate(zip(layer_sizes[:-1], layer_sizes[1:])):
        layer_top_a = v_spacing*(layer_size_a - 1)/2. + (top + bottom)/2.
        layer_top_b = v_spacing*(layer_size_b - 1)/2. + (top + bottom)/2.
        for m in range(layer_size_a):
            for o in range(layer_size_b):
                line = plt.Line2D([n*h_spacing + left, (n + 1)*h_spacing + left],
                                  [layer_top_a - m*v_spacing, layer_top_b - o*v_spacing], c='k')
                ax.add_artist(line)

import numpy as np

def ReLu(x):
    return np.maximum(x,0)

def softmax(z):
    return np.exp(z)/(np.sum(np.exp(z),axis=0))[np.newaxis,:]

def ff_nn_2_ReLu(x, w1, w2, v, b1, b2, c):
    a = np.dot(w1, x) + b1
    h1 = ReLu(a)
    b = np.dot(w2, h1) + b2
    h2 = ReLu(b)
    y = np.dot(v, h2) + c
    return softmax(y)

    df_dx = -2*(1-vector[0])-400*(vector[1]-vector[0]**2)*vector[0]
    df_dy = 200*(vector[1]-vector[0]**2)
    return np.array([df_dx, df_dy])

def grad_descent(starting_point=None, iterations=10, learning_rate=12):
    if starting_point:
        point = starting_point
    else:
        point = np.random.uniform(-1.5,1.5,size=2)
    trajectory = [point]
    
    for i in range(iterations):
        grad = grad_f(point)
        point = point - learning_rate * grad
        trajectory.append(point)
    return np.array(trajectory)

def grad_descent_with_momentum(starting_point=None, iterations=10, alpha=.0005, epsilon=0.00025):
    if starting_point:
        point = starting_point
    else:
        point = np.random.uniform(-1.5,1.5,size=2)
    trajectory = [point]
    v = np.zeros(point.size)
    
    for i in range(iterations):
        grad = grad_f(point)
        v = alpha*v + epsilon*grad
        point = point - v
        trajectory.append(point)
    return np.array(trajectory)

3. BackPop¶

(1).

We first compute the derivitive with respect to the bias and the weight matrix on the output layer.

$$ \frac{\partial L}{\partial c} = \hat y - y $$$$\frac{\partial L}{\partial W_3} = ((\hat y - y)^T H_2out)^T$$

Then, for the second hidden layer,

$$\frac{\partial L}{\partial W_2} = H_1out^T (W_3(\hat y - y)^T )^T \odot Relu^{'}(H_2in) ) = H_1out^T (W_3(\hat y - y)^T )^T \odot I_{H_2in > 0} ) $$$$\frac{\partial L}{\partial b_2} = (W_3(\hat y - y)^T )^T \odot Relu^{'}(H_2in) ) = (W_3(\hat y - y)^T )^T \odot I_{H_2in > 0} ) $$

Then, for the first hidden layer

$$ \frac{\partial L}{\partial W_1} = ((W_2^T(W_3(\hat y - y)^T) \odot Relu^{'}(H_2in))) \odot Relu^{'}(H_1in) X)^T = ((W_2^T(w_3(\hat y - y)^T) \odot I_{H_2in>0})) \odot I_{H_1in>0} X)^T$$$$\frac{\partial L}{\partial b_1} = (W_2^T(W_3(\hat y - y)^T) \odot Relu^{'}(H_2in))) \odot Relu^{'}(H_1in) = (W_2^T(w_3(\hat y - y)^T) \odot I_{H_2in>0})) \odot I_{H_1in>0} $$

def grad_f(weight_vector, x, y):
    x = x.T # x (2,n)
    w_1, w_2, v, b_1, b_2, c = pack_params(weight_vector) 
    z_1 = np.dot(w_1, x) + b_1 #(3,n)
    h_1 = ReLu(z_1) #(3,n)
    z_2 = np.dot(w_2, h_1) + b_2 #(2,n)
    h_2 = ReLu(z_2) #(2,n)
    z_3 = np.dot(v, h_2) + c #(3,n)
    y_hat = softmax(z_3).T #(n, 3)
    
    dv = h_2.dot((y_hat-y)).T
    dc = (y_hat-y).sum(axis = 0).reshape(3,1)
    dw2 = (((y_hat-y).dot(v)).T * (h_2 > 0)).dot(h_1.T)
    db2 = (((y_hat - y).dot(v)).T * (h_2 > 0)).sum(axis=1).reshape(2,1)
    dw1 = ((w_2.T.dot(  (((y_hat-y).dot(v)).T * (h_2 > 0))  )) * (h_1 > 0)).dot(x.T)
    db1 = ((w_2.T.dot(  (((y_hat-y).dot(v)).T * (h_2 > 0))  )) * (h_1 > 0)).sum(axis = 1).reshape(3,1)
    
    derivitive_vector = unpack_params(dw1, dw2, dv, db1, db2, dc)    
    return np.array(derivitive_vector)

# functions to transform weight matrix to vectors and also weight vectors to matrix
def unpack_params(w_1, w_2, v, b_1, b_2, c):
    w1_11, w1_12, w1_21, w1_22, w1_31, w1_32 = w_1[0,0], w_1[0,1], w_1[1,0], w_1[1,1], w_1[2,0], w_1[2,1]
    b1_1, b1_2, b1_3 = np.squeeze(b_1[0]), np.squeeze(b_1[1]), np.squeeze(b_1[2]) 
    w2_11, w2_12, w2_13, w2_21, w2_22, w2_23 = w_2[0,0], w_2[0,1], w_2[0,2], w_2[1,0], w_2[1,1], w_2[1,2]
    b2_1, b2_2 = np.squeeze(b_2[0]), np.squeeze(b_2[1])
    v_11, v_12, v_21, v_22, v_31, v_32 = v[0,0], v[0,1], v[1,0], v[1,1], v[2,0], v[2,1]
    c_1, c_2, c_3 = np.squeeze(c[0]), np.squeeze(c[1]), np.squeeze(c[2])
    return np.array([w1_11, w1_12, w1_21, w1_22, w1_31, w1_32, b1_1, b1_2, b1_3, w2_11, w2_12, w2_13, w2_21, w2_22, w2_23
                     ,b2_1, b2_2, v_11, v_12, v_21, v_22, v_31, v_32, c_1, c_2, c_3])

def pack_params(param_vec):
    w1_11, w1_12, w1_21, w1_22, w1_31, w1_32, b1_1, b1_2, b1_3, \
    w2_11, w2_12, w2_13, w2_21, w2_22, w2_23, b2_1, b2_2, \
    v_11, v_12, v_21, v_22, v_31, v_32, c_1, c_2, c_3 = param_vec
    w_1 = np.array([[w1_11, w1_12],[w1_21, w1_22], [w1_31, w1_32]])
    b_1 = np.array([[b1_1],[b1_2],[b1_3]])
    w_2 = np.array([[w2_11, w2_12, w2_13], [w2_21, w2_22, w2_23]])
    b_2 = np.array([[b2_1], [b2_2]])
    v = np.array([[v_11, v_12],[v_21, v_22], [v_31, v_32]])
    c = np.array([[c_1], [c_2], [c_3]])
    return ((w_1, w_2, v, b_1, b_2, c))

# loss function
def loss(y, y_hat):
    # cross entropy
    tot = y * np.log(y_hat)
    return  -tot.sum()
# Relu function
def ReLu(x):
    return np.maximum(x,0)

# softmax function
def softmax(y):
    return np.exp(y)/(np.sum(np.exp(y),axis=0))[np.newaxis,:]

def grad_descent(x, y, iterations=1000, learning_rate=1e-2):
    point = np.random.uniform(-.1,.1,size=26)
    trajectory = [point]
    losses = [loss(y, ff_nn_2_ReLu(x.T, *pack_params(point)).T)]
    
    for i in range(iterations):
        grad = grad_f(point, x, y)
        point = point - learning_rate * grad
        trajectory.append(point)
        losses.append(loss(y, ff_nn_2_ReLu(x.T, *pack_params(point)).T))
    return (np.array(trajectory), losses)
